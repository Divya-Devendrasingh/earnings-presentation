class RiskScoreCalculator {
    constructor(transaction) {
        this.amount = transaction.amount;
        this.location = transaction.location;
        this.history = transaction.userHistory;
    }
    
    calculateRiskScore() {
        let score = 0;
        
        // Amount-based risk
        if (this.amount > 10000) score += 30;
        else if (this.amount > 5000) score += 15;
        
        // Location anomaly detection
        if (this.isUnusualLocation()) score += 25;
        
        // Historical pattern analysis
        score += this.analyzePatterns();
        
        return Math.min(score, 100);
    }
    
    isUnusualLocation() {
        return !this.history.commonLocations
            .includes(this.location);

}
